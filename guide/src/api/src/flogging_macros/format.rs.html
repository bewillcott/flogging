23//!
24//! # format Macro Impl
25//!
26//! Format the Log function call (Rust code)
27//!
28
29use dyn_fmt::AsStrFormatExt;
30use proc_macro::TokenStream;
31use regex::RegexBuilder;
32
33pub(crate) fn format_impl(fmt_str: &str, msg: TokenStream) -> TokenStream {
34    // println!("msg: {}", &msg);
35
36    let fmt = "let __fmt = format!({});\n".format(&[
37        match process_msg(msg){
38            Some(r_msg) => r_msg,
39            None => return TokenStream::new(),
40        }
41        ]);
42
43    let mut buf = String::new();
44    buf.push_str(&fmt);
45    buf.push_str(fmt_str);
46
47    // let rtn = buf.parse().unwrap_or_default();
48    // println!("{rtn}");
49    // rtn
50    buf.parse().unwrap_or_default()
51}
52///
53/// Check for `object` only `msg`.
54///
55/// If found, provide default format string.
56///
57fn process_msg(msg: TokenStream) -> Option<String> {
58    let text = msg.to_string();
59
60    if text.is_empty() {
61        return None;
62    }
63
64    if text.starts_with('\"') && text.ends_with('\"') {
65        return Some(text);
66    }
67
68    let regex_str = "(?<fmt>\".*\\{.*\\}.*\")(,[\\s]*(?<attrs>.*)*)";
69
70    let re = RegexBuilder::new(regex_str)
71        .dot_matches_new_line(true)
72        .build()
73        .unwrap();
74
75    if re.is_match(&text) {
76        Some(text)
77    } else {
78        let count = text.split(',').count();
79
80        let mut buf = String::new();
81
82        if count == 1 {
83            buf.push_str("\"{}\", ");
84        } else {
85            buf.push_str("\"{}");
86
87            for _i in 1..count {
88                buf.push_str(", {}");
89            }
90
91            buf.push_str("\", ");
92        }
93
94        buf.push_str(&text);
95
96        Some(buf)
97    }
98}
